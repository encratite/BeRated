worker_processes  1;

events {
	worker_connections 1024;
}

http {
	include mime.types;
	default_type application/octet-stream;

	keepalive_timeout 65;

	server {
		listen 80;
		server_name localhost;
		
		location /Static {
			alias X:/BeRated/BeRated/;
		}
		
		location ~ ^/(Images|Scripts|Stylesheets)/ {
			rewrite ^/(.+) /Static/$1;
		}
		
		location = /favicon.ico {
			rewrite .* /Images/Favicon.ico;
		}
		
		location = / {
			rewrite .* /General;
		}
		
		location / {
			proxy_pass http://localhost:0/;
		}
	}
}
